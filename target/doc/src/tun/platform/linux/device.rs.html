37const OVERWRITE_SIZE: usize = std::mem::size_of::<libc::__c_anonymous_ifr_ifru>();

39/// A TUN device using the TUN/TAP Linux driver.
40pub struct Device {
41    tun_name: String,
42    pub(crate) tun: Tun,

44}

58impl Device {
59    /// Create a new `Device` for the given `Configuration`.
60    pub fn new(config: &Configuration) -> Result<Self> {
61        if let Some(fd) = config.raw_fd {
62            let close_fd_on_drop = config.close_fd_on_drop.unwrap_or(true);
63            let tun_fd = Fd::new(fd, close_fd_on_drop)?;
64            let mtu = config.mtu.unwrap_or(crate::DEFAULT_MTU);
65            let packet_information = config.platform_config.packet_information;
66            let tun_name = config.tun_name.clone().unwrap_or_else(|| "".into());
67            let ctl = Fd::new(unsafe { libc::socket(AF_INET, SOCK_DGRAM, 0) }, true)?;
68            return Ok(Device {
69                tun: Tun::new(tun_fd, mtu, packet_information),
70                tun_name,
71                ctl,
72            });
73        }
74
75        let mut device = unsafe {
76            let dev_name = match config.tun_name.as_ref() {
77                Some(tun_name) => {
78                    let tun_name = CString::new(tun_name.clone())?;
79
80                    if tun_name.as_bytes_with_nul().len() > IFNAMSIZ {
81                        return Err(Error::NameTooLong);
82                    }
83
84                    Some(tun_name)
85                }
86
87                None => None,
88            };
89
90            let mut req: ifreq = mem::zeroed();
91
92            if let Some(dev_name) = dev_name.as_ref() {
93                ptr::copy_nonoverlapping(
94                    dev_name.as_ptr() as *const c_char,
95                    req.ifr_name.as_mut_ptr(),
96                    dev_name.as_bytes_with_nul().len(),
97                );
98            }
99
100            let device_type: c_short = config.layer.unwrap_or(Layer::L3).into();
101
102            let queues_num = config.queues.unwrap_or(1);
103            if queues_num != 1 {
104                return Err(Error::InvalidQueuesNumber);
105            }
106
107            let iff_no_pi = IFF_NO_PI as c_short;
108            let iff_multi_queue = IFF_MULTI_QUEUE as c_short;
109            let iff_napi = IFF_NAPI as c_short;
110            let iff_vnet_hdr = IFF_VNET_HDR as c_short;
111            let packet_information = config.platform_config.packet_information;
112            let napi = config.platform_config.napi;
113            let vnet_hdr = config.platform_config.vnet_hdr;
114            req.ifr_ifru.ifru_flags = device_type
115                | if packet_information { 0 } else { iff_no_pi }
116                | if napi { iff_napi } else { 0 }
117                | if vnet_hdr { iff_vnet_hdr } else { 0 }
118                | if queues_num > 1 { iff_multi_queue } else { 0 };
119
120            let tun_fd = {
121                let fd = libc::open(c"/dev/net/tun".as_ptr() as *const _, O_RDWR);
122                let tun_fd = Fd::new(fd, true).map_err(|_| std::io::Error::last_os_error())?;
123
124                if let Err(err) = tunsetiff(tun_fd.inner, &mut req as *mut _ as *mut _) {
125                    return Err(std::io::Error::from(err).into());
126                }
127
128                tun_fd
129            };
130
131            let mtu = config.mtu.unwrap_or(crate::DEFAULT_MTU);
132
133            let ctl = Fd::new(libc::socket(AF_INET, SOCK_DGRAM, 0), true)?;
134
135            let tun_name = CStr::from_ptr(req.ifr_name.as_ptr())
136                .to_string_lossy()
137                .to_string();
138            Device {
139                tun_name,
140                tun: Tun::new(tun_fd, mtu, packet_information),
141                ctl,
142            }
143        };
144
145        if config.platform_config.ensure_root_privileges {
146            device.configure(config)?;
147        }
148
149        Ok(device)
150    }
151
152    /// Prepare a new request.
153    unsafe fn request(&self) -> ifreq {
154        let mut req: ifreq = unsafe { mem::zeroed() };
155        unsafe {
156            ptr::copy_nonoverlapping(
157                self.tun_name.as_ptr() as *const c_char,
158                req.ifr_name.as_mut_ptr(),
159                self.tun_name.len(),
160            )
161        };
162
163        req
164    }

166    /// Make the device persistent.
167    pub fn persist(&mut self) -> Result<()> {
168        unsafe {
169            if let Err(err) = tunsetpersist(self.as_raw_fd(), 1) {
170                Err(std::io::Error::from(err).into())
171            } else {
172                Ok(())
173            }
174        }
175    }

214    /// Recv a packet from tun device, waiting at most `timeout`.
215    ///
216    /// A TUN device is a character device, so `SO_RCVTIMEO`-style read
217    /// timeouts do not apply; this is implemented as poll(2) followed by a
218    /// read on every call. If no packet arrives within `timeout`, an error
219    /// of kind `std::io::ErrorKind::TimedOut` is returned. A zero `timeout`
220    /// checks for a packet without blocking. If several threads receive from
221    /// the same device concurrently, another thread may consume the packet
222    /// first and the read may still block.
223    pub fn recv_timeout(
224        &self,
225        buf: &mut [u8],
226        timeout: std::time::Duration,
227    ) -> std::io::Result<usize> {
228        self.tun.recv_timeout(buf, timeout)
229    }

261impl AbstractDevice for Device {
262    fn tun_index(&self) -> Result<i32> {
263        let name = self.tun_name()?;
264        Ok(posix::tun_name_to_index(name)? as i32)
265    }
266
267    fn tun_name(&self) -> Result<String> {
268        Ok(self.tun_name.clone())
269    }
270
271    fn set_tun_name(&mut self, value: &str) -> Result<()> {
272        unsafe {
273            let tun_name = CString::new(value)?;
274
275            if tun_name.as_bytes_with_nul().len() > IFNAMSIZ {
276                return Err(Error::NameTooLong);
277            }
278
279            let mut req = self.request();
280            ptr::copy_nonoverlapping(
281                tun_name.as_ptr() as *const c_char,
282                req.ifr_ifru.ifru_newname.as_mut_ptr(),
283                value.len(),
284            );
285
286            if let Err(err) = siocsifname(self.ctl.as_raw_fd(), &req) {
287                return Err(std::io::Error::from(err).into());
288            }
289
290            self.tun_name = value.into();
291
292            Ok(())
293        }
294    }
295
296    fn enabled(&mut self, value: bool) -> Result<()> {
297        unsafe {
298            let mut req = self.request();
299
300            if let Err(err) = siocgifflags(self.ctl.as_raw_fd(), &mut req) {
301                return Err(std::io::Error::from(err).into());
302            }
303
304            if value {
305                req.ifr_ifru.ifru_flags |= (IFF_UP | IFF_RUNNING) as c_short;
306            } else {
307                req.ifr_ifru.ifru_flags &= !(IFF_UP as c_short);
308            }
309
310            if let Err(err) = siocsifflags(self.ctl.as_raw_fd(), &req) {
311                return Err(std::io::Error::from(err).into());
312            }
313
314            Ok(())
315        }
316    }
317
318    fn address(&self) -> Result<IpAddr> {
319        unsafe {
320            let mut req = self.request();
321            if let Err(err) = siocgifaddr(self.ctl.as_raw_fd(), &mut req) {
322                return Err(std::io::Error::from(err).into());
323            }
324            let sa = sockaddr_union::from(req.ifr_ifru.ifru_addr);
325            Ok(std::net::SocketAddr::try_from(sa)?.ip())
326        }
327    }
328
329    fn set_address(&mut self, value: IpAddr) -> Result<()> {
330        unsafe {
331            let mut req = self.request();
332            ipaddr_to_sockaddr(value, 0, &mut req.ifr_ifru.ifru_addr, OVERWRITE_SIZE);
333            if let Err(err) = siocsifaddr(self.ctl.as_raw_fd(), &req) {
334                return Err(std::io::Error::from(err).into());
335            }
336            Ok(())
337        }
338    }

399            if let Err(err) = siocsifnetmask(self.ctl.as_raw_fd(), &req) {

406    fn mtu(&self) -> Result<u16> {
407        unsafe {
408            let mut req = self.request();
409
410            if let Err(err) = siocgifmtu(self.ctl.as_raw_fd(), &mut req) {
411                return Err(std::io::Error::from(err).into());
412            }
413
414            req.ifr_ifru
415                .ifru_mtu
416                .try_into()
417                .map_err(|_| Error::TryFromIntError)
418        }
419    }
420
421    fn set_mtu(&mut self, value: u16) -> Result<()> {
422        unsafe {
423            let mut req = self.request();
424            req.ifr_ifru.ifru_mtu = value as i32;
425
426            if let Err(err) = siocsifmtu(self.ctl.as_raw_fd(), &req) {
427                return Err(std::io::Error::from(err).into());
428            }
429            self.tun.set_mtu(value);
430            Ok(())
431        }
432    }
