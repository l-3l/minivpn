1/// Runs a command and returns an error if the command fails, just convenience for users.
2#[doc(hidden)]
3#[allow(dead_code)]
4pub fn run_command(command: &str, args: &[&str]) -> std::io::Result<Vec<u8>> {
5    let full_cmd = format!("{} {}", command, args.join(" "));
6    log::debug!("Running command: \"{full_cmd}\"...");
7    let out = match std::process::Command::new(command).args(args).output() {
8        Ok(out) => out,
9        Err(e) => {
10            log::error!("Run command: \"{full_cmd}\" failed with: {e}");
11            return Err(e);
12        }
13    };
14    if !out.status.success() {
15        let err = String::from_utf8_lossy(if out.stderr.is_empty() {
16            &out.stdout
17        } else {
18            &out.stderr
19        });
20        let info = format!("Run command: \"{full_cmd}\" failed with {err}");
21        log::error!("{info}");
22        return Err(std::io::Error::other(info));
23    }
24    Ok(out.stdout)
25}
